## 安装
```jsx| pure
$ npm install boty-design --save

$ yarn add boty-design
```

## 使用
```jsx| pure
import { Button } from "boty-design";

const App = () => (
     <Button onClick={handlerSyncClick} loading={true}  >
        异步按钮
      </Button>
);
```
